60.WAQTD NAME , ANNUAL SALARY OF THE EMPLOYEES IF THEIR ANNUAL SALARY IS MORE THAN ALL THE SALESMAN

 SELECT ENAME,SAL*12 AS ANNUALSAL
 FROM EMP
 WHERE SAL*12 > ALL (SELECT SAL*12
                     FROM EMP
                     WHERE JOB='SALESMAN');
