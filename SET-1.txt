11.WAQTD NAME OF THE EMPLOYEE ALONG WITH THEIR ANNUAL SALARY.
SQL> select ENAME,SAL*12 AS AnnualSal
  2  FROM EMP;

12.WAQTD ENAME AND JOB FOR ALL THE EMPLOYEE WITH THEIR HALF TERM SALARY.
SQL> SELECT ENAME,JOB,SAL*6 as HalfTermSal	// here  AS  or as SAME 
  2  FROM EMP;

13.WAQTD ALL THE DETAILS OF THE EMPLOYEES ALONG WITH AN ANNUALBONUS OF 2000.
SQL> SELECT EMP.*, 2000 AS ANUALBONUS
  2  FROM EMP;

14.WAQTD NAME SALARY AND SALARY WITH A HIKE OF 10%.

SQL> SELECT ENAME,SAL,SAL+SAL/100*10 AS HIKE			//--- SAL+SAL*0.1 ASLO RIGHT
  2  FROM EMP;				

15. WAQTD NAME AND SALARY WITH DEDUCTION OF 25%.
SQL> SELECT ENAME,SAL,SAL-SAL*0.25	// SAL-SAL/100*25
  2  FROM EMP;


16.WAQTD NAME AND SALARY WITH MONTHLY HIKE OF 50.
SQL> SELECT ENAME,SAL,SAL+50 AS HIKE
  2  FROM EMP;

17.WAQTD NAME AND ANNUAL SALARY WITH DEDUCTION OF 10%.
SQL> SELECT ENAME,SAL*12-SAL*12*0.1 AS REVISED		// SAL*12 -SAL*12/100*10
  2  FROM EMP;


18. WAQTD TOTAL SALARY GIVEN TO EACH EMPLOYEE (SAL+COMM).
 SQL> SELECT ENAME,SAL+COMM
  2  FROM EMP;

19.WAQTD DETAILS OF ALL THE EMPLOYEES ALONG WITH ANNUAL SALARY.
SQL> SELECT EMP.*,SAL*12 AS ANNUALSAL
  2  FROM EMP;

20.WAQTD NAME AND DESIGNATION ALONG WITH 100 PENALTY IN SALARY.  // DESIGNATION MEAN IS JOB
